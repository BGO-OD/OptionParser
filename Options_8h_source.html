143  template <typename T> typename std::enable_if<std::is_integral<T>::value, std::istream&>::type operator>>(std::istream& aStream, postFixedNumber<T>& aNumber) {
144  T n;
145  aStream >> n;
146  if (!aStream.eof()) {
147  auto c = aStream.peek();
148  static std::string gMultipliers("kMGTPEZYRQ");
149  auto m = gMultipliers.find(c);
150  if (m != std::string::npos) {
151  aStream.get(); // dispose multiplier postfix
152  auto n0 = n;
153  for (unsigned i = 0; i <= m; i++) {
154  if (wouldOverflow(n)) {
155  std::string msg("postfix ");
156  msg += c;
157  msg += " to big with prefix ";
158  msg += std::to_string(n0);
159  msg += ", max value is ";
160  msg += std::to_string(std::numeric_limits<T>::max());
161  throw std::overflow_error(msg);
162  }
163  n *= 1024;
164  }
165  }
166  }
167  aNumber = n;
168  return aStream;
169  };
170  template <typename T> typename std::enable_if<std::is_floating_point<T>::value, std::istream&>::type operator>>(std::istream& aStream, postFixedNumber<T>& aNumber) {
171  T n;
172  aStream >> n;
173  if (!aStream.eof()) {
174  auto c = aStream.peek();
175  static std::string gMultipliers("kMGTPEZYRQ");
176  static std::map<const char, double> gDividers{{'d',10.},
177  {'c',100.},
178  {'m',1000.},
179  {'u',1000000.},
180  {'n',1000000000.},
181  {'p',1000000000000.},
182  {'f',1000000000000000.},
183  {'a',1000000000000000000.},
184  {'z',1000000000000000000000.},
185  {'y',1000000000000000000000000.},
186  {'r',1000000000000000000000000000.},
187  {'q',1000000000000000000000000000000.},
188  };
189  auto m = gMultipliers.find(c);
190  if (m != std::string::npos) {
191  aStream.get(); // dispose multiplier postfix
192  auto factor = 1000;
193  if (!aStream.eof()) {
194  c = aStream.peek();
195  if (c == 'i') { // use binary multiplier
196  aStream.get(); // dispose of the 'i'
197  factor = 1024;
198  }
199  }
200  for (unsigned i = 0; i <= m; i++) {
201  n *= factor;
202  }
203  } else {
204  const auto& divider = gDividers.find(c);
205  if (divider != gDividers.end()) {
206  aStream.get(); // dispose divider postfix
207  n = n / divider->second;
208  }
209 
210  }
211  }
212  aNumber = n;
213  return aStream;
214  };

521  virtual void fAddToRange(rangeValueType aValue) {
522  lRange.emplace(aValue);

524  template <typename TT = std::string> typename std::enable_if < (!std::is_same<rangeValueType, std::string>::value) && std::is_same<TT, std::string>::value, void >::type fAddToRange(const TT& aString) {

543  void fAddToRangeFromStream(std::istream& aStream) override {
544  std::string buf;
545  std::getline(aStream, buf);
546  std::stringstream sbuf(buf);
547  while (!sbuf.eof()) {
548  rangeValueType value;
549  using escapedIO::operator>>;
550  sbuf >> std::setbase(0) >> value;
551  fAddToRange(value);
552  }
553  };
554  void fWriteRange(std::ostream &aStream) const override {
555  using escapedIO::operator<<;
556  if (! lRange.empty()) {
557  aStream << "# allowed range is";
558  if (lRange.size() == 2) {
559  aStream << " [" << *(lRange.cbegin()) << "," << *(lRange.crbegin()) << "]\n";
560  } else {
561  aStream << ":";
562  for (const auto& it : lRange) {
563  aStream << " " << it;
564  }
565  aStream << "\n";
566  }
567  }
568  };
569  virtual void fCheckValueForRange(const compareValueType& aValue) const {
570  using escapedIO::operator<<;
571  if (lRange.empty()) {
572  return;
573  } else if (lRange.size() == 2) {
574  if (*(lRange.cbegin()) <= aValue && aValue <= *(lRange.crbegin())) {
575  return;
576  } else {
577  throw typedRangeError<compareValueType>(this, aValue);
578  }
579  } else {
580  for (const auto& it : lRange) {
581  if (it == aValue) {
582  return;
583  }
584  }
585  throw typedRangeError<compareValueType>(this, aValue);
586  }
587  }
